Table 1: users (This stores each account)
- id
- name
- email
- password
- created_at
Notes: id is PK, email should be unique, password should be stored hashed in real life
CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    email VARCHAR(150) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

Table 2: shopping_lists (This stores each list created by a user)
- id
- user_id
- title 
- created_at
Notes: one user can create many lists, user_id is FK to users(id)
CREATE TABLE shopping_lists (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    title VARCHAR(100) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id)
        ON DELETE CASCADE
);
- ON DELETE CASCADE means if a user is deleted, their shopping lists get deleted too.

Table 3: shopping_items (This stores the actual items inside a shopping list)
- id
- list_id
- item_name
- quantity
- category
- bought
- created_at
Notes: one shopping list has many items, list_id is FK to shopping_lists(id), bought can be true/false
CREATE TABLE shopping_items (
    id INT AUTO_INCREMENT PRIMARY KEY,
    list_id INT NOT NULL,
    item_name VARCHAR(100) NOT NULL,
    quantity INT NOT NULL DEFAULT 1,
    category VARCHAR(50),
    bought BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (list_id) REFERENCES shopping_lists(id)
        ON DELETE CASCADE
);

Table 4: pantry_items (This stores items the user already has at home)
- id
- user_id
- item_name
- quantity
- category
- expiry_date
- created_at
Notes: pantry belongs directly to user, user_id is FK to users(id)
CREATE TABLE pantry_items (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    item_name VARCHAR(100) NOT NULL,
    quantity INT NOT NULL DEFAULT 1,
    category VARCHAR(50),
    expiry_date DATE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id)
        ON DELETE CASCADE
);


Table 5: ingredients (This stores reusable ingredient names)
- id
- name
- category
CREATE TABLE ingredients (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL UNIQUE,
    category VARCHAR(50)
);

Table 6: list_ingredients (This is the junction table between lists and ingredients)
==> Because one list can have many ingredients, and one ingredient can appear in many lists.
- id
- list_id
- ingredient_id
- quantity
CREATE TABLE list_ingredients (
    id INT AUTO_INCREMENT PRIMARY KEY,
    list_id INT NOT NULL,
    ingredient_id INT NOT NULL,
    quantity INT NOT NULL DEFAULT 1,
    FOREIGN KEY (list_id) REFERENCES shopping_lists(id)
        ON DELETE CASCADE,
    FOREIGN KEY (ingredient_id) REFERENCES ingredients(id)
        ON DELETE CASCADE
);

Relationships: 
1. User -> ShoppingList (One-to-Many)
2. ShoppingList -> ShoppingItem (One-to-Many)
3. User -> PantryItem (One-to-Many)
